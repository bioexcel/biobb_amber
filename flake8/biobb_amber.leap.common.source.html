72 def _from_string_to_list(input_data: Optional[Union[str, list[str]]]) -> list[str]:
73 """
74 Converts a string to a list, splitting by commas or spaces. If the input is already a list, returns it as is.
75 Returns an empty list if input_data is None.
76  
77 Parameters:
78 input_data (str, list, or None): The string, list, or None value to convert.
79  
80 Returns:
81 list: A list of string elements or an empty list if input_data is None.
82 """
83 if input_data is None:
84 return []
85  
86 if isinstance(input_data, list):
87 # If input is already a list, return it
88 return input_data
89  
90 # If input is a string, determine the delimiter based on presence of commas
91 delimiter = "," if "," in input_data else " "
92 items = input_data.split(delimiter)
93  
94 # Remove whitespace from each item and ignore empty strings
95 processed_items = [item.strip() for item in items if item.strip()]
96  
97 return processed_items
